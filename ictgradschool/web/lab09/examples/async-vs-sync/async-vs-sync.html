Demo of standard synchronous/blocking code

When functionTwo() is called, the program will wait until it finishes.

        let x = undefined;
        function functionOne(){
            x = 1;
            alert("B: " + x);
            functionTwo();
            alert("D: " + x);
        }
        function functionTwo(){
            x = 2;
            alert("C: " + x);
        }
        alert("A: " + x);
        functionOne();
        alert("E: " + x);

Demo of asynchronous/non-blocking code

functionTwo() is asynchronous so the rest of the sequence will continue while it awaits the AJAX response.

        let x = undefined;
        function functionOne(){
            x = 1;
            alert("B: " + x);
            functionTwo();
            alert("D: " + x);
        }
        async function functionTwo(){
            let articleResponseObj = await fetch(`https://sporadic.nz/ajax/articles?id=1`);
            x = 2;
            alert("C: " + x);
        }
        alert("A: " + x);
        functionOne();
        alert("E: " + x);
